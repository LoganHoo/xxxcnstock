## 财务过滤器配置

### `config/filters/financial_filter.yaml`

```yaml
# ============================================
# 财务过滤器配置
# ============================================

filters:
  # ----------------------------------------
  # ROE过滤器
  # ----------------------------------------
  roe_filter:
    enabled: true
    description: "ROE(净资产收益率)过滤器"
    params:
      min_roe: 15.0          # 最低ROE (%)
      max_roe: 100.0         # 最高ROE (%)
    
  # ----------------------------------------
  # 盈利能力过滤器
  # ----------------------------------------
  profitability_filter:
    enabled: true
    description: "盈利能力综合过滤器"
    params:
      min_gross_margin: 30.0     # 最低毛利率 (%)
      min_net_margin: 10.0       # 最低净利率 (%)
      min_roe: 15.0              # 最低ROE (%)
      min_roa: 5.0               # 最低ROA (%)
    
  # ----------------------------------------
  # 成长能力过滤器
  # ----------------------------------------
  growth_filter:
    enabled: true
    description: "成长能力过滤器"
    params:
      min_revenue_growth: 15.0       # 最低营收增长率 (%)
      min_profit_growth: 20.0        # 最低净利润增长率 (%)
      min_asset_growth: 10.0         # 最低总资产增长率 (%)
    
  # ----------------------------------------
  # 偿债能力过滤器
  # ----------------------------------------
  solvency_filter:
    enabled: true
    description: "偿债能力过滤器"
    params:
      max_debt_ratio: 60.0           # 最高资产负债率 (%)
      min_current_ratio: 1.5         # 最低流动比率
      min_quick_ratio: 1.0           # 最低速动比率
    
  # ----------------------------------------
  # 现金流过滤器
  # ----------------------------------------
  cash_flow_filter:
    enabled: true
    description: "现金流过滤器"
    params:
      min_operating_cash_flow: 0     # 最低经营现金流
      require_positive: true         # 是否要求为正
      min_cash_flow_ratio: 0.8       # 最低现金流/净利润比率
    
  # ----------------------------------------
  # 财务综合评分过滤器
  # ----------------------------------------
  financial_composite_filter:
    enabled: true
    description: "财务综合评分过滤器"
    params:
      min_score: 60.0            # 最低综合评分
      top_n: 50                  # 只保留前N名
      weights:                   # 各维度权重
        profitability: 0.40      # 盈利能力
        solvency: 0.20           # 偿债能力
        growth: 0.25             # 成长能力
        cash_flow: 0.15          # 现金流
```

## 配置加载说明

### Python代码中加载配置

```python
import yaml
from pathlib import Path

def load_config(config_name: str) -> dict:
    """加载配置文件"""
    config_path = Path("config") / f"{config_name}.yaml"
    
    with open(config_path, 'r', encoding='utf-8') as f:
        config = yaml.safe_load(f)
    
    return config

# 加载过滤器配置
financial_filter_config = load_config("filters/financial_filter")
market_behavior_config = load_config("filters/market_behavior_filter")

# 获取具体过滤器配置
roe_params = financial_filter_config['filters']['roe_filter']['params']
print(f"ROE最小值: {roe_params['min_roe']}%")
```

### 环境变量替换

```python
import os
import re

def replace_env_vars(config: dict) -> dict:
    """替换配置中的环境变量"""
    config_str = yaml.dump(config)
    
    # 查找 ${VAR_NAME} 格式的变量
    pattern = r'\$\{([^}]+)\}'
    matches = re.findall(pattern, config_str)
    
    for var_name in matches:
        env_value = os.getenv(var_name, '')
        config_str = config_str.replace(f'${{{var_name}}}', env_value)
    
    return yaml.safe_load(config_str)
```
